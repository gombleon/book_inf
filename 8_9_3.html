8.9.3. Деки

Дек или двусторонняя очередь (double-ended queue) — это линейный список, в котором все операции вставки и удаления (и, как правило, операции доступа к данным) выполняются на обоих концах списка.

Пусть узел дека содержит два ссылочных поля, в одном будет храниться ссылка на предыдущий узел, а в другом — ссылка на следующий узел:

TYPE узел=RECORD

name: ARRAY 20 OF CHAR;

prev: указатель

next: указатель

END;

указатель=POINTER TO узел;

В этом случае дек задается начальным узлом first и конечным узлом last:

NEW(first);

first.prev:=NIL;

first.name:'первый узел'

NEW(last);

first.next:=last;

last.next:=NIL;

last.prev:=first;

last.name:='последний узел';

Начальный узел дека в поле prev содержит ссылку на NIL, конечный узел дека в поле next также ссылается на NIL.

Пример 8.9.3.1 создания дека, добавления в дек узла, удаления узла из дека и просмотра содержимого информационного поля дека.

MODULE dec;

IMPORT Log := StdLog, In := i21sysIn, Math;

TYPE список=RECORD

name: ARRAY 20 OF CHAR;

prev:указатель;

next:указатель;

END;

указатель=POINTER TO список;

VAR first, last:указатель;

PROCEDURE new*;

VAR x: указатель;

BEGIN

IF first=NIL THEN

NEW(first);

In.Open;

ASSERT( In.done );

In.String(first.name);

ASSERT( In.done );

first.next:=NIL;

first.prev:=NIL;

last:=first;

Log.String('Создан первый элемент дека.');

Log.Ln

ELSE

Log.String('Очередь уже существует.');

Log.Ln

END;

END new;

PROCEDURE append_first*;

VAR x: указатель;

BEGIN

IF first#NIL THEN

NEW(x);

In.Open;

ASSERT( In.done );

In.String(x.name);

ASSERT( In.done );

x.next:=first;

first.prev:=x;

x.prev:=NIL;

first:=x;

Log.String('Добавлен новый элемент в начало дека.');

Log.Ln

ELSE

new

END;

END append_first;

PROCEDURE append_last*;

VAR x: указатель;

BEGIN

IF first#NIL THEN

NEW(x);

In.Open;

ASSERT( In.done );

In.String(x.name);

ASSERT( In.done );

x.prev:=last;

last:=x;

last.next:=NIL;

Log.String('Добавлен новый элемент в конец дека.');

Log.Ln

ELSE

new

END;

END append_last;

PROCEDURE delete_first*;

VAR

BEGIN

IF first#NIL THEN

first:=first.next;

first.prev:=NIL;

Log.String("Удалили первый элемент дека.");

Log.Ln

ELSE

Log.String("Дек не существует.")

END;

END delete_first;

PROCEDURE delete_last*;

VAR

BEGIN

IF last#NIL THEN

last:=last.prev;

last.next:=NIL;

Log.String("Удалили последний элемент дека.");

Log.Ln

ELSE

Log.String("Дек не существует.");

Log.Ln

END;

END delete_last;

PROCEDURE view1*;

VAR x: указатель;

BEGIN

IF first#NIL THEN

x:=first;

WHILE x.next#NIL DO

Log.String(x.name);

Log.Ln;

x:=x.next

END;

Log.String(x.name);

Log.Ln;

ELSE

Log.String('Дек не существует.');

Log.Ln

END;

END view1;

PROCEDURE view2*;

VAR x: указатель;

BEGIN

IF last#NIL THEN

x:=last;

WHILE x.prev#NIL DO

Log.String(x.name);

Log.Ln;

x:=x.prev

END;

Log.String(x.name);

Log.Ln;

ELSE

Log.String('Дек не существует.');

Log.Ln

END;

END view2;

END dec.
