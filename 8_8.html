8.8. Массивы

Массив — структура, состоящая из некоторого количества элементов, имеющих один и тот же тип, называемый типом элементов. Количество элементов массива называется его длиной. Элементы массива выбираются с помощью индексов, являющихся целыми числами из диапазоне от 0 до длина минус 1.

Объявление массива:

массив: ARRAY [длина {«,» длина}] OF тип_элемента_массива.

Пример 8.8.1. Дан массив из n целых чисел. Найти количество различных чисел среди элементов этого массива. (Число действий должно быть порядка n2).

Решение будет состоять в сравнении каждого i-го числа с остальными n-i числами. Следовательно, для решения задачи потребуется (n-1) + (n-2) + … + 1 = n⁢ (n-1) 2 операций сравнения, из чего следует оценка вычислительной сложности алгоритма решения задачи f⁡(n) ∈ O⁡( n2 ) .

Реализуем алгоритм, например, так:

MODULE massiv1;

IMPORT Log := StdLog, In := i21sysIn, Ran := i21eduRandom;

PROCEDURE do*;

CONST n=7;

VAR a: ARRAY n OF INTEGER;

i, k, j: INTEGER;

BEGIN

FOR i :=0 TO n-1 DO

a[i]:=Ran.Int(100);

Log.Int(a[i])

END;

Log.Ln;

k:=1;

FOR i :=0 TO n-2 DO

j:=i+1;

LOOP

IF (a[i]=a[j]) THEN

EXIT

END;

INC(j);

IF j=n THEN

INC(k);

EXIT

END;

END;

END;

Log.String('Всего различных элементов в массиве ');

Log.Int(k);

Log.Ln

END do;

END massiv1.

massiv1.do

Пространственная сложность алгоритма не обсуждается, поскольку алгоритм не требует использования дополнительного массива.

Пусть теперь массив упорядочен, к примеру, по возрастанию. В этом случае для нахождения количества различных элементов в массиве потребуется всего n-1 операция сравнения, и оценка вычислительной сложности алгоритма будет иметь вид f⁡(n) ∈ O⁡( n ) .

Пример реализации алгоритма для упорядоченного массива:

MODULE massiv2;

IMPORT Log := StdLog, In := i21sysIn, Ran := i21eduRandom;

PROCEDURE do*;

CONST n=10;

VAR a: ARRAY n OF INTEGER;

i, k: INTEGER;

BEGIN

a[0]:=1;

Log.Int(a[0]);

FOR i :=1 TO n-1 DO

a[i]:=a[i-1]+Ran.Int(2);

Log.Int(a[i])

END;

Log.Ln;

k:=1;

FOR i :=0 TO n-2 DO

IF a[i]#a[i+1] THEN

INC(k)

END;

END;

Log.String('Всего различных элементов в массиве ');

Log.Int(k);

Log.Ln

END do;

END massiv2.

massiv2.do

Пример 8.8.2.Дан массив a[0] … a[ n-1 ] целых чисел. Не используя других массивов, переставить элементы массива в обратном порядке.

MODULE massiv3;

IMPORT Log := StdLog, In := i21sysIn, Ran := i21eduRandom;

PROCEDURE do*;

VAR a: POINTER TO ARRAY OF INTEGER;

i, k, n: INTEGER;

BEGIN

In.Open;

ASSERT(In.Done);

In.Int(n);

ASSERT(In.Done & (n>0));

Log.String('Исходный массив:');

Log.Ln;

NEW(a, n);

FOR i :=0 TO n-1 DO

a[i]:=Ran.Int(10);

Log.Int(a[i])

END;

Log.Ln;

i:=0;

WHILE i < n-1-i DO

k:=a[i];

a[i]:=a[n-1-i];

a[n-1-i]:=k;

INC(i)

END;

Log.String('Массив после перестановки:');

Log.Ln;

FOR i :=0 TO n-1 DO

Log.Int(a[i])

END;

Log.Ln;

END do;

END massiv3.

Пример 8.8.3.(Из книги Д. Гриса) Дан массив целых чисел a[0] … a[ n+m-1 ] , рассматриваемый как соединение двух его отрезков: начала a[0] … a[ n-1 ] длины n и конца a[n] … a[ n+m-1 ] длины m. Не используя дополнительных массивов, переставить начало и конец. (Число действий порядка m + n.)

Вариант 1. Сохраним начальный элемент в переменной, затем каждый элемент, кроме последнего, заменим значением следующего элемента, а последний элемент — сохраненным в переменной значением начального элемента. Повторим операцию n-1 раз.

MODULE massiv4;

IMPORT Log := StdLog, In := i21sysIn, Ran := i21eduRandom;

PROCEDURE do*;

VAR a: POINTER TO ARRAY OF INTEGER;

i, k, n, m: INTEGER;

BEGIN

In.Open;

ASSERT(In.Done);

In.Int(n);

ASSERT(In.Done & (n>0));

In.Int(m);

ASSERT(In.Done & (m>0));

NEW(a, n+m);

FOR i :=0 TO n+m-1 DO

a[i]:=Ran.Int(10);

END;

Log.Ln;

Log.String('Исходный массив:');

Log.Ln;

FOR i :=0 TO n-1 DO

Log.Int(a[i])

END;

Log.Tab;

FOR i :=n TO m+n-1 DO

Log.Int(a[i])

END;

Log.Ln;

FOR i :=0 TO n-1 DO

k:=a[0];

FOR j :=0 TO n+m-2 DO

a[j]:=a[j+1]

END;

a[m+n-1]:=k;

END;

Log.String('Массив после перестановки:');

Log.Ln;

FOR i :=0 TO m-1 DO

Log.Int(a[i])

END;

Log.Tab;

FOR i :=m TO m+n-1 DO

Log.Int(a[i])

END;

Log.Ln;

END do;

END massiv4.

Вариант 2. Перевернем (расположим в обратном порядке) отдельно начало и конец массива, а затем перевернем весь массив как единое целое.

MODULE massiv5;

IMPORT Log := StdLog, In := i21sysIn, Ran := i21eduRandom;

PROCEDURE do*;

VAR a: POINTER TO ARRAY OF INTEGER;

i, k, n, m: INTEGER;

BEGIN

In.Open;

ASSERT(In.Done);

In.Int(n);

ASSERT(In.Done & (n>0));

In.Int(m);

ASSERT(In.Done & (m>0));

NEW(a, n+m);

FOR i :=0 TO n+m-1 DO

a[i]:=Ran.Int(10);

END;

Log.Ln;

Log.String('Исходный массив:');

Log.Ln;

FOR i :=0 TO n-1 DO

Log.Int(a[i])

END;

Log.Tab;

FOR i :=n TO m+n-1 DO

Log.Int(a[i])

END;

Log.Ln;

i:=0;

WHILE i < n-1-i DO

k:=a[i];

a[i]:=a[n-1-i];

a[n-1-i]:=k;

INC(i)

END;

Log.Ln;

i:=n;

WHILE i < n+m DIV 2 DO

k:=a[i];

a[i]:=a[2*n+m-1-i];

a[2*n+m-1-i]:=k;

INC(i)

END;

i:=0;

WHILE i < m+n-1-i DO

k:=a[i];

a[i]:=a[m+n-1-i];

a[m+n-1-i]:=k;

INC(i)

END;

Log.String('Массив после перестановки:');

Log.Ln;

FOR i :=0 TO m-1 DO

Log.Int(a[i])

END;

Log.Tab;

FOR i :=m TO m+n-1 DO

Log.Int(a[i])

END;

Log.Ln;

END do;

END massiv5.
