8.12.3. Рекурсивное порождение комбинаторных объектов

Рассмотрим примеры рекурсивного порождения комбинаторных объектов из [41], с. 122 — 126.

Пример 8.12.3.1. Написать программу, которая печатает по одному разу все последовательности длины n, составленные из чисел 1…k (их количество равно kn).

MODULE seq1;

IMPORT Log := StdLog, In := i21sysIn;

PROCEDURE do*;

CONST n=2;k=3;

VAR a: ARRAY n OF INTEGER;

t, i: INTEGER;

PROCEDURE generate;

VAR i, j: INTEGER;

BEGIN

IF t=n-1 THEN

FOR i :=0 TO n-1 DO

Log.Int(a[i]);

Log.Tab

END;

Log.Ln;

ELSE

INC(t);

FOR j:=1 TO k DO

a[t]:=j;

generate;

END;

DEC(t);

END;

END generate;

BEGIN

In.Open;

ASSERT(In.Done);

t:=-1;

generate;

END do;

END seq1.

Пример 8.12.3.2. Написать программу, которая печатала бы все перестановки чисел 1…n по одному разу.

MODULE seq2;

IMPORT Log:=StdLog, In:=i21sysIn;

PROCEDURE do*;

CONST n=4;

VAR a: ARRAY n OF INTEGER;

t, i: INTEGER;

PROCEDURE generate;

VAR i, j: INTEGER;

BEGIN

IF t=n-1 THEN

FOR i:=0 TO n-1 DO

Log.Int(a[i]);

Log.Tab

END;

Log.Ln;

ELSE

FOR j:=t+1 TO n-1 DO

i:=a[t+1];

a[t+1]:=a[j];

a[j]:=i;

INC(t);

generate;

DEC(t);

i:=a[t+1];

a[t+1]:=a[j];

a[j]:=i;

END;

END;

END generate;

BEGIN

FOR i:=0 TO n-1 DO

a[i]:=i+1

END;

t:=-1;

generate;

END do;

END seq2.

Демонстрация алгоритма решения примера 8.12.3.2:

Введите n

Пример 8.12.3.3. Напечатать все возрастающие последовательности длины k, элементами которых являются натуральные числа от 1 до n. (Предполагается, что k≤n, иначе таких последовательностей не существует.)

MODULE seq3;

IMPORT Log := StdLog, In := i21sysIn;

PROCEDURE do*;

CONST n=4;k=3;

VAR a: ARRAY n OF INTEGER;

t, i, j: INTEGER;

PROCEDURE generate;

VAR i: INTEGER;

BEGIN

IF t=k THEN

FOR i :=1 TO k DO

Log.Int(a[i]);

Log.Tab

END;

Log.Ln;

ELSE

INC(t);

FOR i:=a[t-1]+1 TO t-k+n DO

a[t]:=i;

generate;

END;

DEC(t)

END;

END generate;

BEGIN

t:=1;

FOR j:=1 TO 1-k+n DO

a[1]:=j;

generate

END;

END do;

END seq3.

Демонстрация алгоритма решения примера 8.12.3.3:

Введите n
Введите k

Пример 8.12.3.4. Перечислить все представления положительного целого числа n в виде суммы последовательности невозрастающих целых положительных слагаемых.

MODULE seq4;

IMPORT Log:= StdLog, In:= i21sysIn;

PROCEDURE do*;

CONST n=4;

VAR a: ARRAY n OF INTEGER;

t, i, s, j: INTEGER;

PROCEDURE generate;

VAR i: INTEGER;

BEGIN

IF s=n THEN

FOR i :=0 TO t DO

Log.String('a[');Log.Int(i);Log.String(']=');Log.Int(a[i]);

Log.Tab

END;

Log.Ln;

ELSE

FOR i:=1 TO MIN(a[t], n-s) DO

INC(t);

a[t]:=i;

s:=s+i;

generate;

DEC(s);

DEC(t);

END;

END;

END generate;

BEGIN

t:=0;

Log.Clear;

FOR j:=1 TO n DO

a[t]:=j;

s:=j;

generate

END;

END do;

END seq4.

Демонстрация алгоритма решения примера 8.12.3.4:

Введите n
